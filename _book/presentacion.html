# Definir los datos
dt <- mtcars[1:5, 1:6]

# Crear una tabla con m√∫ltiples estilos
kbl(dt, booktabs = TRUE, 
    caption = "Tabla de Ejemplo con Estilos") %>%
  kable_styling(
    bootstrap_options = c("striped", "hover", "condensed"),
    full_width = FALSE,
    position = "center",
    font_size = 10
  ) %>%
  add_header_above(c(" ", "Grupo 1" = 3, "Grupo 2" = 3)) %>%
  column_spec(1, bold = TRUE, color = "red",
              background = "yellow") %>%
  column_spec(2, width = "5em") %>%
  row_spec(0, bold = TRUE, color = "white",
           background = "#D7261E") %>%
  row_spec(1:5, background = "#F7F7F7")

Explicaci√≥n de los Estilos

  • booktabs: Utiliza el paquete booktabs para mejorar el dise√±o de la tabla.

  • bootstrap_options: A√±ade varias opciones de estilo como striped (filas alternas de color), hover (resaltar filas al pasar el mouse) y condensed (espaciado reducido).

  • full_width: Ajusta la tabla para que no ocupe todo el ancho disponible.

  • position: Define la posici√≥n de la tabla, en este caso centra la tabla en la p√°gina.

  • font_size: Cambia el tama√±o de la fuente de la tabla.

  • add_header_above: A√±ade un encabezado adicional encima de la tabla para agrupar columnas.

  • column_spec: Especifica estilos para columnas individuales. En este caso, la primera columna es negrita, roja y con fondo amarillo, y la segunda columna tiene un ancho espec√≠fico.

  • row_spec: Especifica estilos para filas individuales. La fila de encabezado (0) es negrita, blanca con fondo rojo oscuro, y las filas 1 a 5 tienen un fondo gris claro.

5.2.1 Tabla din√°mica con estilos

# Cargar los datos
data("mtcars")

# Crear una tabla din√°mica con personalizaci√≥n adicional

          options = list(
            pageLength = 5,  # N√∫mero de filas por p√°gina
            autoWidth = TRUE,  # Ajuste autom√°tico del ancho de las columnas
            dom = 'Bfrtip',  # botones, filtro y paginaci√≥n
            buttons = c('copy', 'csv',
                        'excel', 'pdf',
                        'print'), # Botones de exportaci√≥n
            initComplete = JS(
              "function(settings, json) {",
              "$(this.api().table().header()).css({'background-color':
              'skyblue', 'color': 'black'});",
              "}"
            ),
            className = 'cell-border stripe'  # Clases CSS para las celdas
          ),
          class = 'cell-border stripe') %>%
  formatStyle(
    'mpg',  # Columna a estilizar
    backgroundColor = styleInterval(c(20, 30), 
                                    c('yellow', 'orange',
                                      'skyblue'))
  )

5.2.2 Cambio idioma de la tabla

library(DT)
library(htmlwidgets)

# Cargar los datos
data("mtcars")

# Crear una tabla din√°mica con opciones de 
# estilo personalizadas y configuraciones en espa√±ol
datatable(mtcars, 
          extensions = 'Buttons',  # Botones de exportaci√≥n disponibles
          options = list(
            pageLength = 5,  # N√∫mero de filas por p√°gina
            autoWidth = TRUE,  # Ajuste autom√°tico del ancho de las columnas
            dom = 'Bfrtip',  # Botones, filtro y paginaci√≥n
            buttons = list('copy', 'csv', 'excel',
                           'pdf', 'print'),  # Opciones de exportaci√≥n
            language = list(
              url = 
                '//cdn.datatables.net/plug-ins/1.10.25/i18n/Spanish.json'  
            ), # Agregar un archivo json de configuraci√≥n del idioma
            initComplete = JS(
              "function(settings, json) {",
              "$(this.api().table().header()).css({'background-color': 
              '#D7261E', 'color': 'black'});",
              "}"
            )
          ),
          class = 'cell-border stripe') %>%
  formatStyle(
    'mpg',  # Columna a estilizar
    backgroundColor = styleInterval(c(20, 30), c('yellow', 
                                                 'orange', 'red'))
  ) %>%
  htmlwidgets::onRender("
    function(el, x) {
      $(el).find('.dt-buttons').appendTo(
      $(el).closest('.dataTables_wrapper').find('.col-md-6:eq(0)') );
    }
  ")
